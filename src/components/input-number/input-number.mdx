# InputNumber 数字输入框

## 基本用法 Basic usage

<InputNumber
  defaultValue={10}
  onChange={(value) => {
    console.log(value)
  }}
/>

```typescript jsx
import React, { FC } from 'react'
import InputNumber from './input-number'

const App: FC = () => {
  return (
    <InputNumber
      defaultValue={10}
      onChange={(value) => {
        console.log(value)
      }}
    />
  )
}

export default App
```
